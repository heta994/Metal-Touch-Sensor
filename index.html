Metal Touch Sensor Arduino Project

This Arduino project reads both analog voltage and digital touch input from a metal touch sensor.

Arduino Code


int AInput = A0; // Analog input from sensor
int DInput = 3;  // Digital input from sensor

void setup() {
  pinMode(AInput, INPUT);
  pinMode(DInput, INPUT);
  Serial.begin(9600);
}

void loop() {
  float analogValue = analogRead(AInput) * (5.0 / 1023.0);
  int digitalValue = digitalRead(DInput);

  Serial.print("Analog voltage value: ");
  Serial.print(analogValue);
  Serial.print(" V, Touch input: ");

  if (digitalValue == HIGH) {
    Serial.println("detected");
  } else {
    Serial.println("not detected");
  }

  Serial.println("---------------------");
  delay(250);
}
